Commands :

	mkdir git

	cd git

	git init

	touch code.txt

	git add code.txt

	git commit -m "Commiting code.txt."

	git branch develop

	git branch feature

	git branch feature2

	git checkout feature

	touch f1.txt

	git add f1.txt

	git commit -m "Commiting feature1 file."

	git checkout feature2

	touch f2.txt

	git add f2.txt

	git commit -m "Commiting feature2 file."

	git checkout develop

	git merge feature

	git merge feature2

	git checkout master

	git merge develop

	git remote add origin "https://github.com/ShubhShri198/CASE-STUDY---GIT-WORKFLOW"

	git push origin '*:*'
